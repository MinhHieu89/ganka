  <name>Task 2: Replace SessionWarningModal Dialog with AlertDialog</name>
  <files>
    frontend/src/features/auth/components/SessionWarningModal.tsx
  </files>
  <action>
    Replace the current Dialog-based SessionWarningModal with AlertDialog:

    Current implementation uses Dialog with workarounds:
    - `onOpenChange={() => {}}` (prevent close)
    - `onPointerDownOutside={(e) => e.preventDefault()}`
    - `onEscapeKeyDown={(e) => e.preventDefault()}`
    - `hideCloseButton` prop

    AlertDialog handles all of this by default (it's designed for non-dismissible confirmations).

    New implementation:
    ```typescript
    import {
      AlertDialog,
      AlertDialogContent,
      AlertDialogDescription,
      AlertDialogFooter,
      AlertDialogHeader,
      AlertDialogTitle,
    } from "@/shared/components/AlertDialog"
    import { Button } from "@/shared/components/Button"

    export function SessionWarningModal({ open, remainingSeconds, onExtend, onLogout }: Props) {
      return (
        <AlertDialog open={open}>
          <AlertDialogContent>
            <AlertDialogHeader>
              <div className="flex items-center justify-center mb-2">
                <div className="flex items-center justify-center w-12 h-12 bg-destructive/10 text-destructive">
                  <IconClock className="h-6 w-6" />
                </div>
              </div>
              <AlertDialogTitle className="text-center">
                {t("session.expiring")}
              </AlertDialogTitle>
              <AlertDialogDescription className="text-center">
                {t("session.expiringMessage", { minutes: timeDisplay })}
              </AlertDialogDescription>
            </AlertDialogHeader>
            <div className="flex items-center justify-center py-4">
              <span className="text-4xl font-mono font-bold tabular-nums text-destructive">
                {timeDisplay}
              </span>
            </div>
            <AlertDialogFooter className="flex-row gap-2 sm:justify-center">
              <Button variant="outline" onClick={onLogout} className="flex-1">
                <IconLogout className="h-4 w-4 mr-2" />
                {t("session.logout")}
              </Button>
              <Button onClick={onExtend} className="flex-1">
                {t("session.extend")}
              </Button>
            </AlertDialogFooter>
          </AlertDialogContent>
        </AlertDialog>
      )
    }
    ```

    Key changes:
    - AlertDialog instead of Dialog (semantically correct for non-dismissible)
    - Remove all manual prevention workarounds (onOpenChange, onPointerDownOutside, onEscapeKeyDown, hideCloseButton)
    - AlertDialogContent/Header/Title/Description/Footer instead of Dialog equivalents
    - Use Button from wrapper (not AlertDialogAction/AlertDialogCancel -- we need custom onClick handlers, not form submission)
    - Keep the clock icon, countdown timer, and time formatting logic
    - Keep i18n translations
    - Import from AlertDialog and Button wrappers

    Note: AlertDialog does NOT render a close X button by default and prevents ESC dismissal, which is exactly what we need.
