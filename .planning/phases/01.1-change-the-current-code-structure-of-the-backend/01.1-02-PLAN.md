```csharp
public static IEndpointRouteBuilder MapAuditApiEndpoints(this IEndpointRouteBuilder app)
{
    var group = app.MapGroup("/api/audit").RequireAuthorization();

    group.MapGet("/logs", async ([AsParameters] GetAuditLogsQuery query, IMessageBus bus) =>
    {
        var result = await bus.InvokeAsync<GetAuditLogsResponse>(query);
        return Results.Ok(result);
    });

    // ... similar for export and access logs
    // Preserve exact URL paths from existing endpoints
    // Preserve authorization requirements ([Authorize] -> RequireAuthorization())

    return app;
}
```
