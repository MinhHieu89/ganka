   ```csharp
   // Idempotency: skip items already billed from this dispensing event
   var existingSourceIds = invoice.LineItems
       .Where(li => li.SourceType == "Dispensing" && li.SourceId == @event.VisitId)
       .Select(li => li.Description)
       .ToHashSet();

   foreach (var item in @event.Items)
   {
       if (existingSourceIds.Contains(item.DrugName)) continue;
       invoice.AddLineItem(...);
   }
   ```

3. **HandleGlassesOrderCreated.cs (CR-06):** Add idempotency check:
   ```csharp
   // Idempotency: skip items already billed from this glasses order
   var existingSourceIds = invoice.LineItems
       .Where(li => li.SourceType == "GlassesOrder" && li.SourceId == @event.OrderId)
       .Select(li => li.Description)
       .ToHashSet();

   foreach (var item in @event.Items)
   {
       if (existingSourceIds.Contains(item.Description)) continue;
       invoice.AddLineItem(...);
   }
   ```
