       ```csharp
       public static class HandleDrugPrescriptionAddedHandler
       {
           public static async Task Handle(
               DrugPrescriptionAddedIntegrationEvent @event,
               IInvoiceRepository invoiceRepository,
               IServiceCatalogRepository serviceCatalogRepository,
               IBillingNotificationService notificationService,
               IUnitOfWork unitOfWork,
               ILogger logger,
               CancellationToken ct)
           {
               // Get or create invoice (same pattern as HandleDrugDispensed)
               var invoice = await invoiceRepository.GetByVisitIdAsync(@event.VisitId, ct);
               if (invoice is null)
               {
                   var number = await invoiceRepository.GetNextInvoiceNumberAsync(DateTime.UtcNow.Year, ct);
                   invoice = Invoice.Create(number, @event.PatientId, @event.PatientName,
                       @event.VisitId, new BranchId(@event.BranchId));
                   invoiceRepository.Add(invoice);
               }

               // Idempotency: skip items already billed from prescriptions
               var existingPrescriptionItems = invoice.LineItems
                   .Where(li => li.SourceType == "Prescription" && li.SourceId == @event.VisitId)
                   .Select(li => li.Description)
                   .ToHashSet();

               foreach (var item in @event.Items)
               {
                   if (existingPrescriptionItems.Contains(item.DrugName))
                       continue;

                   // Look up price from ServiceCatalog using DrugCatalogItemId
                   // If not found, use 0 (cashier can manually adjust)
                   decimal unitPrice = 0m;
                   // Try to find price via service catalog or direct lookup
                   // For now, the price lookup needs the Pharmacy module's selling price
                   // Since we cannot directly query Pharmacy from Billing, we use ServiceCatalog
                   // or accept 0 price (line items with 0 price can be updated by cashier)

                   invoice.AddLineItem(
                       item.DrugName,
                       null, // DrugNameVi not available from prescription
                       unitPrice,
                       item.Quantity,
                       Department.Pharmacy,
                       @event.VisitId,
                       "Prescription");
               }

               await unitOfWork.SaveChangesAsync(ct);

               // Send SignalR notifications
               foreach (var item in @event.Items)
               {
                   if (!existingPrescriptionItems.Contains(item.DrugName))
                   {
                       await notificationService.NotifyLineItemAddedAsync(
                           invoice.Id, invoice.InvoiceNumber, item.DrugName,
                           0m, "Pharmacy", ct);
                   }
               }
           }
       }
       ```
