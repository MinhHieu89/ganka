2. Add a new method `RemoveLineItemsBySource(Guid sourceId, string sourceType)` to `Invoice.cs`:
   ```csharp
   public void RemoveLineItemsBySource(Guid sourceId, string sourceType)
   {
       EnsureDraft();
       var matching = _lineItems.Where(li => li.SourceId == sourceId && li.SourceType == sourceType).ToList();
       foreach (var item in matching)
           _lineItems.Remove(item);
       if (matching.Count > 0)
       {
           RecalculateTotals();
           SetUpdatedAt();
       }
   }
   ```
   This method is used by the event handler to remove prescription items when doctor removes a prescription.
