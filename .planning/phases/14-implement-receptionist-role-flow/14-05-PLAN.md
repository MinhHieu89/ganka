    export const Route = createFileRoute("/_authenticated/patients/intake")({
      beforeLoad: () => requirePermission("Patient.Create"),
      component: PatientIntakePage,

    function PatientIntakePage() {
      // Check for patientId search param (edit mode from check-in incomplete flow)
      const { patientId } = Route.useSearch()
      // If patientId: load patient data, pass as defaultValues
      return <PatientIntakeForm patientId={patientId} mode={patientId ? "edit" : "create"} />
    }
