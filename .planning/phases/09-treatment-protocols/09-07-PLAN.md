    Create Treatment.Infrastructure/Configurations/ directory and:

    1. TreatmentProtocolConfiguration.cs:
       - IEntityTypeConfiguration of TreatmentProtocol
       - ToTable("TreatmentProtocols")
       - HasKey(x => x.Id)
       - Name: IsRequired, HasMaxLength(200)
       - TreatmentType: HasConversion of int
       - PricingMode: HasConversion of int
       - DefaultPackagePrice: HasColumnType("decimal(18,2)")
       - DefaultSessionPrice: HasColumnType("decimal(18,2)")
       - CancellationDeductionPercent: HasColumnType("decimal(5,2)")
       - DefaultParametersJson: HasMaxLength(4000)
       - Description: HasMaxLength(1000), nullable
       - BranchId conversion: HasConversion(b => b.Value, v => new BranchId(v))
       - Index on TreatmentType
       - Index on IsActive

    2. TreatmentPackageConfiguration.cs:
       - IEntityTypeConfiguration of TreatmentPackage
       - ToTable("TreatmentPackages")
       - HasKey(x => x.Id)
       - PatientName: IsRequired, HasMaxLength(200)
       - TreatmentType, Status, PricingMode: HasConversion of int
       - PackagePrice, SessionPrice: HasColumnType("decimal(18,2)")
       - ParametersJson: HasMaxLength(4000)
       - BranchId conversion
       - Navigation(x => x.Sessions).UsePropertyAccessMode(PropertyAccessMode.Field)
       - Navigation(x => x.Versions).UsePropertyAccessMode(PropertyAccessMode.Field)
       - HasMany(Sessions) -> WithOne() -> HasForeignKey(TreatmentPackageId) -> OnDelete(Cascade)
       - HasMany(Versions) -> WithOne() -> HasForeignKey(TreatmentPackageId) -> OnDelete(Cascade)
       - HasOne(CancellationRequest) configuration -- optional one-to-one
       - Ignore computed: SessionsCompleted, SessionsRemaining, IsComplete
       - Indexes: PatientId, Status, TreatmentType, ProtocolTemplateId
