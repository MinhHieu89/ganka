**Step 2: Update GetDashboardStatsHandler**

Modify `GetDashboardStats.cs` to add `IMessageBus` (from `Wolverine`) as a handler parameter. Wolverine auto-injects it. Update the handler to:

```csharp
using Patient.Application.Interfaces;
using Shared.Domain;
using Wolverine;
using Scheduling.Contracts.Queries;    // GetTodayAppointmentCountQuery
using Clinical.Contracts.Dtos;         // GetActiveVisitCountQuery
using Treatment.Contracts.Queries;     // GetActiveTreatmentCountQuery

// ... existing query/DTO records stay the same ...

public static class GetDashboardStatsHandler
{
    public static async Task<Result<DashboardStatsDto>> Handle(
        GetDashboardStatsQuery query,
        IPatientRepository patientRepository,
        IMessageBus bus,
        CancellationToken cancellationToken)
    {
        var totalPatients = await patientRepository.GetActiveCountAsync(cancellationToken);

        // Cross-module queries via Wolverine message bus
        int todayAppointments = 0;
        int activeVisits = 0;
        int activeTreatments = 0;

        try { todayAppointments = await bus.InvokeAsync<int>(new GetTodayAppointmentCountQuery(), cancellationToken); } catch { /* graceful degradation */ }
        try { activeVisits = await bus.InvokeAsync<int>(new GetActiveVisitCountQuery(), cancellationToken); } catch { /* graceful degradation */ }
        try { activeTreatments = await bus.InvokeAsync<int>(new GetActiveTreatmentCountQuery(), cancellationToken); } catch { /* graceful degradation */ }

        var stats = new DashboardStatsDto(
            TotalPatients: totalPatients,
            TodayAppointments: todayAppointments,
            ActiveVisits: activeVisits,
            ActiveTreatments: activeTreatments);

        return stats;
    }
}
```

NOTE: The `using` statements for contracts depend on where the query records were placed in Task 1. If `GetActiveVisitCountQuery` was added to `Clinical.Contracts.Dtos` namespace, use that. Adjust namespaces accordingly.
