public async Task<List<Icd10Code>> SearchAsync(string term, int limit = 50, CancellationToken ct = default)
{
    return await _context.Icd10Codes
        .Where(c =>
            c.Code.Contains(term) ||

**Bug 2: Field names not localized (VisitAmendmentHistory.tsx line 87)**

Create a `formatFieldLabel` function in VisitAmendmentHistory.tsx that maps raw field keys to localized display names using the `t()` function from react-i18next.

The field keys follow these patterns:
- `refraction.{type}.{field}` (e.g., "refraction.manifest.odSph")
- `examinationNotes`
- `diagnosis.added.{code}` / `diagnosis.removed.{code}`

Build the formatter:
```typescript
function formatFieldLabel(fieldKey: string, t: TFunction): string {
  // Refraction fields: "refraction.manifest.odSph" -> "Manifest OD SPH"
  const refractionMatch = fieldKey.match(/^refraction\.(\w+)\.(\w+)$/)
  if (refractionMatch) {
    const [, type, field] = refractionMatch
    const typeLabel = t(`refraction.${type}`) // "Manifest" / "Autorefraction" / "Cycloplegic"
    const fieldLabel = REFRACTION_FIELD_LABELS[field] || field
    return `${typeLabel} ${fieldLabel}`
  }

  // Refraction type-level: "refraction.manifest" -> "Manifest"
  const typeMatch = fieldKey.match(/^refraction\.(\w+)$/)
  if (typeMatch) {
    return t(`refraction.${typeMatch[1]}`)
  }

  // Examination notes
  if (fieldKey === "examinationNotes") return t("visit.examinationNotes")

  // Diagnosis: "diagnosis.added.H40.1" -> "Diagnosis Added: H40.1"
  if (fieldKey.startsWith("diagnosis.added.")) {
    const code = fieldKey.replace("diagnosis.added.", "")
    return `${t("visit.diagnosis")} (+) ${code}`
  }
  if (fieldKey.startsWith("diagnosis.removed.")) {
    const code = fieldKey.replace("diagnosis.removed.", "")
    return `${t("visit.diagnosis")} (-) ${code}`
  }

  return fieldKey // Fallback: show raw key
}
```
