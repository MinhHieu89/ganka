  **Step 2 — SignOffSection.tsx: Compute actual diff at re-sign time**
  When the visit is in Amended status and the user clicks Sign Off (re-sign), compute the diff between the stored baseline and the current visit state.

  Add a function `computeFieldChanges` that:
  1. Finds the latest amendment in visit.amendments
  2. Parses its fieldChangesJson as the baseline (VisitBaseline)
  3. Compares each field in the baseline to the current visit state
  4. Returns only the fields that actually changed as `Array<{ field: string; oldValue: string; newValue: string }>`

  ```typescript
  function computeFieldChanges(visit: VisitDetailDto): string {
    const latestAmendment = [...visit.amendments].sort(
      (a, b) => new Date(b.amendedAt).getTime() - new Date(a.amendedAt).getTime()
    )[0]

    if (!latestAmendment) return "[]"

    let baseline: VisitBaseline
    try {
      baseline = JSON.parse(latestAmendment.fieldChangesJson)
    } catch {
      return "[]"
    }

    // If fieldChangesJson is already an array (old format), skip diff
    if (Array.isArray(baseline)) return latestAmendment.fieldChangesJson

    const changes: Array<{ field: string; oldValue: string; newValue: string }> = []

    // Compare examination notes
    const currentNotes = visit.examinationNotes ?? null
    if (baseline.examinationNotes !== currentNotes) {
      changes.push({
        field: "examinationNotes",
        oldValue: baseline.examinationNotes ?? "-",
        newValue: currentNotes ?? "-",
      })
    }

    // Compare refractions by type
    const refractionTypeLabel = (type: number) =>
      type === 0 ? "manifest" : type === 1 ? "autorefraction" : "cycloplegic"

    for (const baseRef of baseline.refractions) {
      const currentRef = visit.refractions.find(r => r.type === baseRef.type)
      if (!currentRef) {
        changes.push({
          field: `refraction.${refractionTypeLabel(baseRef.type)}`,
          oldValue: JSON.stringify(baseRef),
          newValue: "(removed)",
        })
        continue
      }
      // Compare each field
      const refFields = ["odSph","odCyl","odAxis","osSph","osCyl","osAxis","odAdd","odPd","osAdd","osPd","ucvaOd","ucvaOs","bcvaOd","bcvaOs","iopOd","iopOs","axialLengthOd","axialLengthOs"] as const
      for (const f of refFields) {
        const oldVal = baseRef[f]
        const newVal = currentRef[f as keyof typeof currentRef]
        if (oldVal !== newVal) {
          changes.push({
            field: `refraction.${refractionTypeLabel(baseRef.type)}.${f}`,
            oldValue: oldVal != null ? String(oldVal) : "-",
            newValue: newVal != null ? String(newVal) : "-",
          })
        }
      }
    }

    // Check for new refractions added during amendment
    for (const curRef of visit.refractions) {
      if (!baseline.refractions.find(r => r.type === curRef.type)) {
        changes.push({
          field: `refraction.${refractionTypeLabel(curRef.type)}`,
          oldValue: "(none)",
          newValue: "(added)",
        })
      }
    }

    // Compare diagnoses
    const baselineDx = new Set(baseline.diagnoses.map(d => `${d.icd10Code}:${d.laterality}`))
    const currentDx = new Set(visit.diagnoses.map(d => `${d.icd10Code}:${d.laterality}`))

    for (const key of baselineDx) {
      if (!currentDx.has(key)) {
        changes.push({ field: "diagnosis", oldValue: key, newValue: "(removed)" })
      }
    }
    for (const key of currentDx) {
      if (!baselineDx.has(key)) {
        changes.push({ field: "diagnosis", oldValue: "(none)", newValue: key })
      }
    }

    return JSON.stringify(changes)
  }
  ```

        params: { path: { visitId } },
        body: fieldChangesJson ? { visitId, fieldChangesJson } : { visitId },
