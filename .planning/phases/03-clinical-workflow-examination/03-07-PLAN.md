    3. **Build baseline snapshot on submit:** When the user submits the amendment reason, build a `fieldChangesJson` that captures the current visit state as the amendment baseline. The amendment workflow in this app works as follows:
       - Doctor clicks "Amend" on a signed visit
       - AmendmentDialog opens, doctor enters reason
       - On submit, the backend transitions visit to Amended status (editable)
       - Doctor then makes changes via individual auto-save endpoints
       - There is no "finalize amendment" step

    Since the amendment is created BEFORE edits are made (it unlocks the visit), the field-level diff cannot show before/after at creation time. Instead, capture a meaningful snapshot of the visit state at the moment of amendment initiation:

    ```typescript
    const buildFieldChangesSnapshot = (visit: VisitDetailDto): string => {
      const changes: Array<{ fieldName: string; oldValue: string; newValue: string }> = []

      // Capture current state as baseline for audit trail
      if (visit.examinationNotes) {
        changes.push({
          fieldName: "examinationNotes",
          oldValue: visit.examinationNotes,
          newValue: "pending_amendment",
        })
      }

      if (visit.refractions.length > 0) {
        for (const r of visit.refractions) {
          const typeLabel = r.refractionType === 0 ? "manifest" : r.refractionType === 1 ? "autorefraction" : "cycloplegic"
          changes.push({
            fieldName: `refraction.${typeLabel}`,
            oldValue: JSON.stringify({
              odSph: r.odSph, odCyl: r.odCyl, odAxis: r.odAxis,
              osSph: r.osSph, osCyl: r.osCyl, osAxis: r.osAxis,
            }),
            newValue: "pending_amendment",
          })
        }
      }

      if (visit.diagnoses.length > 0) {
        changes.push({
          fieldName: "diagnoses",
          oldValue: JSON.stringify(visit.diagnoses.map(d => ({
            code: d.icd10Code,
            laterality: d.laterality,
          }))),
          newValue: "pending_amendment",
        })
      }

      // If no specific data to snapshot, provide a generic baseline
      if (changes.length === 0) {
        changes.push({
          fieldName: "visit",
          oldValue: "signed_state",
          newValue: "amendment_initiated",
        })
      }

      return JSON.stringify(changes)
    }
    ```

    ```typescript
    await amendMutation.mutateAsync({
      visitId,

      fieldChangesJson,
    })
    ```
