    ```tsx
    useEffect(() => {
      form.reset({
        fullName: patient.fullName,
        phone: patient.phone,
        dateOfBirth: patient.dateOfBirth ? new Date(patient.dateOfBirth) : null,
        gender: patient.gender,
        address: patient.address ?? "",
    ```
