    **1. Add 401 response interceptor to api-client.ts (user requirement: proper token refresh interceptor)**

    Add an `onResponse` handler to the authMiddleware in api-client.ts that:
    - Detects 401 status responses
    - Calls `silentRefresh()` from auth-api.ts to get a new token via HTTP-only cookie
    - Updates the auth store with the new access token
    - Retries the original request with the new token
    - If refresh also fails (e.g., cookie expired), clears auth and redirects to /login
    - Uses a flag/promise to prevent concurrent refresh attempts (if multiple 401s arrive simultaneously, only one refresh call should be made)

    Implementation pattern:
    ```typescript
    import { silentRefresh } from "@/features/auth/api/auth-api"
    import { useAuthStore } from "@/shared/stores/authStore"

    let refreshPromise: Promise<LoginResponse | null> | null = null

    const authMiddleware: Middleware = {
      async onRequest({ request }) {
        const token = useAuthStore.getState().accessToken
        if (token) {
          request.headers.set("Authorization", `Bearer ${token}`)
        }
        return request
      },
      async onResponse({ response, request }) {
        if (response.status === 401) {
          // Avoid concurrent refresh attempts
          if (!refreshPromise) {
            refreshPromise = silentRefresh().finally(() => { refreshPromise = null })
          }
          const result = await refreshPromise
          if (result) {
            useAuthStore.getState().setAuth(
              { id: result.user.id, email: result.user.email, fullName: result.user.fullName, permissions: result.user.permissions, preferredLanguage: result.user.preferredLanguage },
              result.accessToken
            )
            // Retry original request with new token
            const retryRequest = request.clone()
            retryRequest.headers.set("Authorization", `Bearer ${result.accessToken}`)
            return fetch(retryRequest)
          } else {
            // Refresh failed - redirect to login
            useAuthStore.getState().clearAuth()
            window.location.href = "/login"
          }
        }
        return response
      },
    }
    ```

    NOTE: The `silentRefresh` function in auth-api.ts already exists as a standalone async function (not a hook) - perfect for use in middleware. Check if the `LoginResponse` type needs to be imported.
